    <script>
        let currentRegionFilter = 'all';
        let calendar;
        let selectedEventId = null;

        // Dữ liệu gốc từ ảnh
        let promotions = [
            { id: '1', title: 'Reply 1996 - Merchandising', start: '2026-05-04', end: '2026-05-18', region: 'Nationwide', service: 'Merchandising', description: 'Áp dụng cho hóa đơn trên 800k: Tote Bag, T-Shirt, Bag Charm.', color: '#3b82f6' },
            { id: '2', title: 'School Party Combo', start: '2026-05-04', end: '2026-06-01', region: 'Nationwide', service: 'Dine-in & Deli', description: 'Enable School Party Combo cho học sinh.', color: '#3b82f6' },
            { id: '3', title: 'Mothers Day', start: '2026-05-10', end: '2026-05-11', region: 'Nationwide', service: 'Dine-in', description: 'Tặng 1 món tráng miệng cho mẹ (nhóm gia đình).', color: '#10b981' },
            { id: '4', title: 'Golden Week (South)', start: '2026-05-18', end: '2026-05-23', region: 'South', service: 'Dine-in', description: 'Giảm 30% steak cho hóa đơn từ 300k. Chỉ sau 4pm.', color: '#f97316' },
            { id: '5', title: 'Reply 1996 - Photobooth', start: '2026-05-18', end: '2026-05-25', region: 'Nationwide', service: 'Dine-in', description: 'Chụp ảnh photobooth cho hóa đơn từ 500k.', color: '#3b82f6' },
            { id: '6', title: 'Lucky Draw Big Event', start: '2026-05-04', end: '2026-05-27', region: 'Nationwide', service: 'Dine-in', description: 'Iphone 17 pro, GoPro, Garmin... D-Day 26/5.', color: '#ef4444' }
        ];

        document.addEventListener('DOMContentLoaded', function() {
            const calendarEl = document.getElementById('calendar');
            calendar = new FullCalendar.Calendar(calendarEl, {
                initialView: 'dayGridMonth',
                initialDate: '2026-05-01',
                headerToolbar: { left: 'prev,next today', center: 'title', right: 'dayGridMonth,timeGridWeek,listMonth' },
                locale: 'vi',
                firstDay: 1,
                events: promotions,
                editable: true,
                eventClick: (info) => openEditModal(info.event),
                eventDrop: () => updateDashboard(),
                eventResize: () => updateDashboard(),
                height: 700
            });
            calendar.render();
            updateDashboard();
        });

        function filterByRegion(region) {
            currentRegionFilter = region;
            const buttons = document.querySelectorAll('.region-filter-btn');
            buttons.forEach(btn => {
                btn.classList.remove('active', 'bg-white', 'shadow-sm');
                btn.classList.add('text-slate-500');
                if(btn.innerText === (region === 'all' ? 'Tất cả' : region)) {
                    btn.classList.add('active', 'bg-white', 'shadow-sm');
                    btn.classList.remove('text-slate-500');
                }
            });
            applyFilters();
        }

        function applyFilters() {
            const service = document.getElementById('serviceFilter').value;
            
            // Xóa hết sự kiện hiện tại và add lại theo filter
            calendar.removeAllEvents();
            
            const filtered = promotions.filter(p => {
                const matchRegion = currentRegionFilter === 'all' || p.region === currentRegionFilter;
                const matchService = service === 'all' || p.service.includes(service);
                return matchRegion && matchService;
            });
            
            calendar.addEvents(filtered);
            updateDashboard();
        }

        function toggleModal(modalId) {
            const modal = document.getElementById(modalId);
            modal.classList.toggle('hidden');
            modal.classList.toggle('modal-active');
            if (modal.classList.contains('hidden')) {
                document.getElementById('eventForm').reset();
                selectedEventId = null;
                document.getElementById('deleteBtn').classList.add('hidden');
                document.getElementById('modalTitle').innerText = "Tạo chiến dịch mới";
            }
        }

        function openEditModal(event) {
            selectedEventId = event.id;
            document.getElementById('modalTitle').innerText = "Chỉnh sửa chiến dịch";
            document.getElementById('m_title').value = event.title;
            document.getElementById('m_start').value = event.startStr.split('T')[0];
            document.getElementById('m_end').value = event.endStr ? event.endStr.split('T')[0] : event.startStr.split('T')[0];
            document.getElementById('m_region').value = event.extendedProps.region;
            document.getElementById('m_service').value = event.extendedProps.service;
            document.getElementById('m_desc').value = event.extendedProps.description;
            document.getElementById('deleteBtn').classList.remove('hidden');
            toggleModal('eventModal');
        }

        document.getElementById('eventForm').onsubmit = function(e) {
            e.preventDefault();
            const data = {
                id: selectedEventId || String(Date.now()),
                title: document.getElementById('m_title').value,
                start: document.getElementById('m_start').value,
                end: document.getElementById('m_end').value,
                region: document.getElementById('m_region').value,
                service: document.getElementById('m_service').value,
                description: document.getElementById('m_desc').value,
            };

            // Gán màu theo Region
            data.color = data.region === 'South' ? '#f97316' : (data.region === 'North' ? '#a855f7' : '#3b82f6');

            if (selectedEventId) {
                const idx = promotions.findIndex(p => p.id === selectedEventId);
                promotions[idx] = data;
            } else {
                promotions.push(data);
            }

            applyFilters();
            toggleModal('eventModal');
        };

        document.getElementById('deleteBtn').onclick = function() {
            if (confirm('Xóa chiến dịch này?')) {
                promotions = promotions.filter(p => p.id !== selectedEventId);
                applyFilters();
                toggleModal('eventModal');
            }
        };

        function updateDashboard() {
            const currentEvents = calendar.getEvents();
            document.getElementById('totalEvents').innerText = currentEvents.length;
            
            const listContainer = document.getElementById('eventSummaryList');
            listContainer.innerHTML = '';

            currentEvents.forEach(ev => {
                const item = document.createElement('div');
                item.className = "p-4 bg-slate-50 border border-slate-100 hover:border-orange-200 transition-all cursor-pointer rounded-2xl group";
                item.onclick = () => { calendar.gotoDate(ev.start); openEditModal(ev); };
                item.innerHTML = `
                    <div class="flex justify-between items-start mb-2">
                        <span class="text-[9px] font-black px-2 py-0.5 rounded-full uppercase tracking-tighter" style="background-color: ${ev.backgroundColor}22; color: ${ev.backgroundColor}">
                            ${ev.extendedProps.region}
                        </span>
                        <span class="text-[10px] text-slate-400 font-bold">${new Date(ev.start).toLocaleDateString('vi-VN', {day:'2-digit', month:'2-digit'})}</span>
                    </div>
                    <h4 class="text-xs font-black text-slate-700 group-hover:text-orange-600 transition-colors">${ev.title}</h4>
                    <div class="flex items-center gap-1 mt-2 text-[10px] text-slate-400 italic">
                        <svg xmlns="http://www.w3.org/2000/svg" width="10" height="10" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M6 2L3 6l3 4"></path><path d="M10 6H3"></path></svg>
                        ${ev.extendedProps.service}
                    </div>
                `;
                listContainer.appendChild(item);
            });
        }
